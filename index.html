import gensim
from nltk.stem import WordNetLemmatizer
from nltk.corpus import stopwords
import nltk

# Unduh stopwords dan wordnet NLTK (hanya perlu sekali)
nltk.download('stopwords')
nltk.download('wordnet')

stemmer = WordNetLemmatizer()
stop_words = stopwords.words('english')  # Jika butuh stopwords Bahasa Indonesia bisa custom

def lemmatize_stemming(text):
    return stemmer.lemmatize(text, pos='v')

def preprocess(text):
    result = []
    for token in gensim.utils.simple_preprocess(str(text)):
        if token not in stop_words and len(token) > 3:
            result.append(lemmatize_stemming(token))
    return result
